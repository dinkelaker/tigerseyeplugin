793                 * ambiguity value combinations have been visited.
794                 */
795                public boolean nextAmbiguitySmallIncrement() {
796                        if (DEBUG)
797                                System.out.println("nextAmbiguitySmallIncrement before [" + this.smallIncreamentAmbiguityItemIndex
798                                                + "]");
799                        if (DEBUG)
800                                System.out.println("nextAmbiguitySmallIncrement before itemIndexToValueIndex="
801                                                + this.itemIndexToValueIndex);
802 
803                        if (!this.hasNext())
804                                return false;
805 
806                        Integer oldValueIndex = this.itemIndexToValueIndex.get(this.smallIncreamentAmbiguityItemIndex);
807                        if (oldValueIndex == null)
808                                oldValueIndex = 0;
809                        int sizeOfCurrentAmbiguityItem = this.getValueSizeForItemIndex(this.smallIncreamentAmbiguityItemIndex);
810                        boolean nextSmallIncrementItemBelowLimit = (oldValueIndex + 1 < sizeOfCurrentAmbiguityItem);
811                        if (nextSmallIncrementItemBelowLimit) {
812                                // there are still more values for the small increment
813                                for (int i = 0; i < this.smallIncreamentAmbiguityItemIndex; i++) {
814                                        this.itemIndexToValueIndex.put(i, 0);
815                                }
816                                assert (oldValueIndex + 1 < this.ambiguityArray.get(this.smallIncreamentAmbiguityItemIndex).size());
817                                this.itemIndexToValueIndex.put(this.smallIncreamentAmbiguityItemIndex, oldValueIndex + 1);
818                        } else {
819                                while (!nextSmallIncrementItemBelowLimit
820                                                && (this.smallIncreamentAmbiguityItemIndex < this.ambiguityArray.size() - 1)) {
821                                        // no more value for this item, but for next in the small
822                                        // increment
