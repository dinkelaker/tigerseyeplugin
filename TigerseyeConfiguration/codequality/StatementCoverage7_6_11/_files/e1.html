611    /**
612     * TODO move functionalities of this method solely responsible for
613     * extracting keywords to separate class. It is probable that this might be
614     * useful for another client.
615     * 
616     * This method gets a list of methods from an external class from
617     * class-path.
618     * 
619     * @param language
620     *            fill the table with keywords of this language
621     */
622    private void setDeclaredMethodsList(DSLDefinition language) {
623 
624        declaredKeywordsTable.removeAll();
625 
626        // Read all public declared Fields from external class
627        KeyWordExtractor keyWordExtractor = new KeyWordExtractor(
628                language.loadClass());
629        Field[] publicDeclaredFields = keyWordExtractor
630                .getDeclaredLiteralKeywords();
631        for (Field declaredField : publicDeclaredFields) {
632            TableItem tableItem = null;
633 
634            // Add new keyword to table
635            tableItem = new TableItem(declaredKeywordsTable, SWT.BORDER);
636 
637            tableItem.setText(0, declaredField.getName());
638            tableItem.setText(1, declaredField.getType().getSimpleName());
639            tableItem.setText(2, "(native literal)");
640 
641        }
642 
643        Method[] publicDeclaredMethods = keyWordExtractor.getMethodKeywords();
644 
645        // Read all getter setter from external class
646        Set<String> fieldAccessors = new HashSet<String>();
647        Set<String> fieldAccessorTypes = new HashSet<String>();
648 
649        if (publicDeclaredMethods.length == 0) {
650            TableItem tableItem = new TableItem(declaredKeywordsTable,
651                    SWT.BORDER);
652 
653            tableItem.setText(0, "NoMethodsRetrieved");
654            tableItem.setText(1, "");
655        }
656 
657            for (Method declaredMethod : publicDeclaredMethods) {
658 
659                // replace the Java coding conventions prefix from the keyword
660                // name
661                if (declaredMethod.getName().startsWith("get")) {
662                    fieldAccessors.add(declaredMethod.getName().substring(3));
663                    fieldAccessorTypes.add(declaredMethod.getReturnType()
664                            .getName());
665                }
666                if (declaredMethod.getName().startsWith("set")) {
667                    fieldAccessors.add(declaredMethod.getName().substring(3));
668                    fieldAccessorTypes.add(declaredMethod.getReturnType()
669                            .getName());
670                }
671                if (declaredMethod.getName().startsWith("is")) {
672                    if ((declaredMethod.getReturnType().isPrimitive() && (declaredMethod
673                            .getReturnType() == boolean.class))
674                            || (declaredMethod.getReturnType() == Boolean.class)) {
675                        fieldAccessors.add(declaredMethod.getName()
676                                .substring(2));
677                        fieldAccessorTypes.add(declaredMethod.getReturnType()
678                                .getName());
679                    }
680                }
681            }
682 
683            Iterator<String> it = fieldAccessors.iterator();
684            Iterator<String> itTypes = fieldAccessorTypes.iterator();
685            while (it.hasNext() && itTypes.hasNext()) {
686                String item = it.next();
687                String itemType = itTypes.next();
688 
689                TableItem tableItem = null;
690 
691                // Add new keyword to table
692                tableItem = new TableItem(declaredKeywordsTable, SWT.BORDER);
693 
694                tableItem.setText(0, item);
695                tableItem.setText(1, itemType);
696                tableItem.setText(2, "(literal defined by accessor)");
697            }
698 
699        // Read all public declared methods from external class
700            for (Method declaredMethod : publicDeclaredMethods) {
701                TableItem tableItem = null;
702 
703                // Add new keyword to table
704                tableItem = new TableItem(declaredKeywordsTable, SWT.BORDER);
705 
706                tableItem.setText(0, declaredMethod.getName());
707                tableItem.setText(1, declaredMethod.getReturnType()
708                        .getSimpleName());
709 
710                StringBuilder parameters = new StringBuilder();
711                for (int i = 0; i < declaredMethod.getParameterTypes().length; i++) {
712                    if (i > 0) {
713                        parameters = parameters.append(", ");
714                    }
715                    parameters = parameters.append(declaredMethod
716                            .getParameterTypes()[i].getSimpleName().toString());
717                }
718 
719                tableItem.setText(2, parameters.toString());
720            }
721 
722    }
