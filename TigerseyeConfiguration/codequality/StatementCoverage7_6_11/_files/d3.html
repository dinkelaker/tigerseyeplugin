432        private static String fixEmptyVariables(TemplateBuffer buffer, String[] variables) throws MalformedTreeException, BadLocationException {
433                IDocument doc= new Document(buffer.getString());
434                int nLines= doc.getNumberOfLines();
435                MultiTextEdit edit= new MultiTextEdit();
436                Set< Integer > removedLines = new HashSet<Integer>();
437                for (int i= 0; i < variables.length; i++) {
438            TemplateVariable position = findVariable(buffer, variables[i]);
439            /* look if Javadoc tags have to be added */
440                        if (position == null || position.getLength() > 0) {
441                                continue;
442                        }
443                        int[] offsets= position.getOffsets();
444                        for (int k= 0; k < offsets.length; k++) {
445                                int line= doc.getLineOfOffset(offsets[k]);
446                                IRegion lineInfo= doc.getLineInformation(line);
447                                int offset= lineInfo.getOffset();
448                                String str= doc.get(offset, lineInfo.getLength());
449                                if (Strings.containsOnlyWhitespaces(str) && nLines > line + 1 && removedLines.add(new Integer(line))) {
450                                        int nextStart= doc.getLineOffset(line + 1);
451                                        edit.addChild(new DeleteEdit(offset, nextStart - offset));
452                                }
453                        }
454                }
455                edit.apply(doc, 0);
456                return doc.get();
457        }
